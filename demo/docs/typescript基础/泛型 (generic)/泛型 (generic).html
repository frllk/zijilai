在静态编写的时候并不确定传入的参数到底是什么类型，只有当在运行时传入参数后我们才能确定。
// T在传入时才能决定类型
function getFrist<T>(n1: T , n2: T) : T {
  return n1 
}

// 多参数
function getFrist<T,S>(n1: T , n2: S) : T {
  return n1 
}

// 泛型变量
function getArrayLength<T>(arg: Array<T>) {
  console.log((arg as Array<any>).length) // ok
  return arg
}

// 泛型接口
interface IGetFrist<T> {
  (n1: T, n2: T): T
}
const getFrist: IGetFrist<number> = (n1, n2)  {
  return n1
}

  }

// 泛型约束
type Params = number | string
function getFrist<T extends Params>(n1: T , n2: T) : T {
  return n1 
}

  return obj[key]

// 多重泛型约束
interface IConfig1 {
  host: string,
}
interface IConfig2 {
  port: number
}
class Config<T extends IConfig1 & IConfig2> {
  private config : T
  constructor(config:T) {
    this.config = config
  }
  getHost() {
    return this.config.host
  }
  getPort() {
    return this.config.port
  }
}

// 类型守卫
class Config1  {
  host= 'localhost';
  port= 9327;
}

class Config2 {
  url= 'localhost:9527'
}

function getUrl(arg: Config1 | Config2) {
  if(arg instanceof Config1) {
    return arg.host + arg.port
  }else {
    return arg.url
  }
}
// 或者
function getUrl2(arg: Config1 | Config2) {
  if('url' in arg ) {
    return arg.url
    
  }else {
    return arg.host + arg.port
  }
}

// 字面量类型守卫
type Config1 = {
  kind: 'host',
  host: string
  port: number
}

type Config2 = {
  kind: 'url',
  url: string
}

function getUrl(arg: Config1 | Config2) {
  if (arg.kind !== 'url') {
    return arg.host + arg.port
  } else {
    return arg.url
  }
}
