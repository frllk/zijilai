Nulll

Symbol

// Symbol 表示独一无二的值，最大的用法是用来定义对象的唯一属性名
// 编译配置中需要  "lib": ["es6"]
const id = Symbol('key1')
id === Symbol('key1') // false
let syObject = {};
syObject[id] = "kk";
console.log(syObject);    // {Symbol(key1): "kk"}

BigInt 

// BigInt 安全存储最大整数 不受Number的限制
// 编译配置中需要  "lib": ["ESNext"]
const num： bigint = 123

any 可能任何类型

// any 可能任何类型
let a : any = 4
a = 'abc' // 不报错

unknown 可能任何类型

// unknown 可能任何类型 当没有被确定前不能被实例化 getter 作为函数执行
let value : unknown
value.abc // NG
value() // NG
new value() // NG
value[0][1] // NG

never 永不存在类型

// never 永不存在类型
function error(msg: string) : never {
  throw new Error(message)
}
const empty : never[] = []

元组 （Tuple）

// 元组 （Tuple）
const x : [string, number]
x = ['x' , 1] //OK
x = ['x'] //NG
x = ['x' , 1, 2] //NG

object 表示非原始类型

// Object object 表示非原始类型，也就是除 number，string，boolean，symbol，null 或 undefined 之外的类型。
const ob : object = Date
const ob : object = {}
