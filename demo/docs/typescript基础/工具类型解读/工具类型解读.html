// Partial 将属性全部变为可选
type Partial<T> = { [P in keyof T]?: T[P] };

interface User {
    name: string,
    id:number
}
type U = Partial<User>
const user: U = {
    name:''
}

// DeepPartial
interface Company {
    id: number
    name: string
}

interface Person {
    id: number
    name: string
    adress: string
    company: Company
}

type DeepPartial<T> = {
    [U in keyof T]?: T[U] extends object
    ? DeepPartial<T[U]>
    : T[U]
};
type R2 = DeepPartial<Person>

// 关键字
// + /-  给属性添加修饰符
// 传入的属性变为必选项
type Required<T> = { [P in keyof T]-?: T[P] };

// Omit<T, K>的作用是忽略T中的某些属性.
type Omit<T, K> = Pick<T, Exclude<keyof T, K>>
type Foo = Omit<{name: string, age: number}, 'name'> // -> { age: number }

// Merge合并
type O1 = {
    name: string
    id: number
}

type O2 = {
    id: number
    from: string
}
type Compute<A extends any> =
    A extends Function
    ? A
    : { [K in keyof A]: A[K] }
type Merge<O1 extends object, O2 extends object> =
    Compute<O1 & Omit<O2, keyof O1>>
type R2 = Merge<O1, O2>

// Intersection 取T的属性,此属性同样也存在与U
type Props = { name: string; age: number; visible: boolean };
type DefaultProps = { age: number };
type Intersection<T extends object, U extends object> = Pick<
  T,
  Extract<keyof T, keyof U> & Extract<keyof U, keyof T>
>;
// Expect: { age: number; }
type DuplicatedProps = Intersection<Props, DefaultProps>;

// Overwrite属性 用U的属性覆盖T的相同属性
type Props = { name: string; age: number; visible: boolean };
type NewProps = { age: string; other: string };
type Overwrite<
  T extends object,
  U extends object,
  I = Diff<T, U> & Intersection<U, T>
> = Pick<I, keyof I>;
// Expect: { name: string; age: string; visible: boolean; }
type ReplacedProps = Overwrite<Props, NewProps>
  
// Mutable T 的所有属性的 readonly 移除
  type Mutable<T> = {
  -readonly [P in keyof T]: T[P]
}
