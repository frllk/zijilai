keyof

索引类型的查询操作符
interface User  {
    id:string
    name: string
}
function getProps<T>(obj: T, name: keyof T) {
    return obj[name]
}
const user : User = {
    id:'1',
    name: 'tom'
}

getProps(user, 'name') // OK
getProps(user, 'abc') // NG

function pick<T, K extends keyof T>(o: T, names: K[]): T[K][] {
    return names.map(n => o[n]);
}

const res = pick(user, ['name', 'id', ])

条件类型

declare  function func<T extends boolean>(x: T) : T extends true ? string : number

const x : number = func(false)
const y : string = func(true)
const z : string | number = func(Math.random() > 0.5)

// 与联合类型配合
// 裸类型参数,没有被任何其他类型包裹即T
type NakedUsage<T> = T extends boolean ? "YES" : "NO"
// 类型参数被包裹的在元组内即[T]
type WrappedUsage<T> = [T] extends [boolean] ? "YES" : "NO";

type Distributed = NakedUsage<number | boolean> //  = NakedUsage<number> | NakedUsage<boolean> =  "NO" | "YES"
type NotDistributed = WrappedUsage<number | boolean > // "NO"

// 要找出T类型中U不包含的部分
type Diff<T, U> = T extends U ? never : T;
type R = Diff<"a" | "b" | "c" | "d", "a" | "c" | "f">;  // "b" | "d"

// 取出函数类型的属性名称
interface User {
    id: number;
    name: string;
    getStatus(): void;
}
type FunctionPropertyNames<T> = { [K in keyof T]: T[K] extends Function ? K : never }[keyof T]
type R = FunctionPropertyNames<User>;

infer推断类型

表示在 extends 条件语句中待推断的类型变量
// 判断是否是函数类型 如果是则返回参数 如果不是返回泛型T
type method = (s: string) => boolean
type data = string | number
type ParamType<T> = T extends (param: infer P) => any ? P : T;
type param = ParamType<method> // string
type param2 = ParamType<data> // string | number

// 判断是否是函数类型 如果是则返回参数 如果不是返回any
type Func = () => string
type Num = string | boolean
type ReturnType2<T> = T extends (...args: any[]) => infer P ? P : any;
type T1 = ReturnType2<Func> // string
type T2 = ReturnType2<Num> // any

// tuple转union,比如[string, number] -> string | number
type ElementOf<T> = T extends Array<infer E> ? E : never;
type TTuple = [string, number,boolean];
type ToUnion = ElementOf<TTuple>; // string | number

// union 转 intersection，如：string | number -> string & number
type UnionToIntersection = (U extends any ? (k: U) => void : never) extends ((k: infer I) => void) ? I : never;
type Result = UnionToIntersection<string | number>;
